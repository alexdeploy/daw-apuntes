  director VARCHAR(50),

-- Oracle
CREATE TABLE venta (
  cod_emp INTEGER NOT NULL,
  dni VARCHAR(9),

  fecha DATE constraint ven_fec_VN NOT NULL, -- el constraint de NOT NULL es el �nico que no se puede poner abajo como el resto
